        btn.className = `quote-filter-btn px-4 py-2 rounded-full text-sm font-medium transition-colors duration-200 ${category === 'All' ? 'active bg-amber-600 text-white' : 'bg-white text-gray-600 hover:bg-amber-100'}`;

        btn.textContent = categoryText.replace('design', 'Design & UX').replace('engineering', 'Engineering');

        btn.dataset.filter = category;

        btn.addEventListener('click', () => {

            document.querySelectorAll('.quote-filter-btn').forEach(b => b.classList.remove('active', 'bg-amber-600', 'text-white'));

            btn.classList.add('active', 'bg-amber-600', 'text-white');

            filterQuotes(category);

        });

        quoteFilters.appendChild(btn);

    });

    const renderQuotes = (quotes) => {

        quotesGrid.innerHTML = '';

        quotes.forEach(quote => {

            const quoteEl = document.createElement('div');

            quoteEl.className = 'quote-card bg-white p-6 rounded-lg shadow-md flex flex-col h-full';

            quoteEl.dataset.category = quote.category;

            quoteEl.innerHTML = `

                <p class="text-lg italic text-gray-700 flex-grow">“${quote.text}”</p>

                <div class="mt-4 pt-4 border-t border-gray-200">

                    <p class="font-semibold text-gray-800">${quote.who}</p>

                    <p class="text-sm text-gray-500">${quote.role}</p>

                     <a href="${quote.url}" target="_blank" rel="noopener noreferrer" class="text-sm text-amber-700 hover:underline mt-1 inline-block">Source 🔗</a>

                </div>

            `;

            quotesGrid.appendChild(quoteEl);

        });

    };

    const filterQuotes = (category) => {

        const filteredQuotes = category === 'All' ? db.quotes : db.quotes.filter(q => q.category === category);

        renderQuotes(filteredQuotes);

    };

    renderQuotes(db.quotes);

   

    // --- Commandments ---

    const commandmentsAccordion = document.getElementById('commandments-accordion');

    db.commandments.forEach((cmd, index) => {

        const item = document.createElement('div');

        item.className = 'bg-white rounded-lg shadow-md overflow-hidden';

        item.innerHTML = `

            <h2>

                <button class="commandment-title flex justify-between items-center w-full p-5 font-medium text-left text-gray-700 hover:bg-amber-50 focus:outline-none" aria-expanded="false">

                    <span>${cmd.title}</span>

                    <span class="icon-plus text-amber-700 text-2xl font-light">+</span>

                </button>

            </h2>

            <div class="commandment-content hidden">

                <div class="p-5 border-t border-gray-200">

                    <p class="text-gray-600">${cmd.content}</p>

                </div>

            </div>

        `;

        commandmentsAccordion.appendChild(item);

    });

   

    commandmentsAccordion.addEventListener('click', (e) => {

        const title = e.target.closest('.commandment-title');

        if (title) {

            const content = title.parentElement.nextElementSibling;

            const isExpanded = title.getAttribute('aria-expanded') === 'true';

           

            title.setAttribute('aria-expanded', !isExpanded);

            content.classList.toggle('hidden');

        }

    });

   

    // --- Mobile Menu ---

    const mobileMenuButton = document.getElementById('mobile-menu-button');

    const mobileMenu = document.getElementById('mobile-menu');

    mobileMenuButton.addEventListener('click', () => {

        mobileMenu.classList.toggle('hidden');

    });

    // --- Nav scroll highlighting ---

    const sections = document.querySelectorAll('section');

    const navLinks = document.querySelectorAll('.nav-link');

    window.addEventListener('scroll', () => {

        let current = '';

        sections.forEach(section => {

            const sectionTop = section.offsetTop;

            if (pageYOffset >= sectionTop - 70) {

                current = section.getAttribute('id');

            }

        });

        navLinks.forEach(link => {

            link.classList.remove('active');

            if (link.getAttribute('href') === `#${current}`) {

                link.classList.add('active');

            }

        });

    });

});
